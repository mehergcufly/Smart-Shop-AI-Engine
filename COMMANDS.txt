# 1. Clone / navigate to project
cd D:\ssai

# 2. Create Python 3.11 virtual environment
py -3.11 -m venv venv
venv\Scripts\Activate.ps1

# 3. Install all dependencies
pip install -r backend/requirements.txt

# 4. Create PostgreSQL database
psql -U postgres
# Inside psql:
CREATE DATABASE sshop_db;
\q

# 5. Fill in .env file (already done — verify it exists)
cat .env

# 6. Train the ML model (ONE TIME ONLY)
python train_model.py
# Output: backend/models/random_forest.pkl + scaler.pkl

# 7. Initialize database tables
cd backend
python database.py
# Output: ✅ All tables created / verified in sshop_db

# 8. Start the API server
uvicorn main:app --host 0.0.0.0 --port 8000 --reload

# 9. Open dashboard
# Double-click frontend/index.html in File Explorer
# OR visit: http://localhost:8000/app

# 10. Test API
# Visit: http://localhost:8000/docs
